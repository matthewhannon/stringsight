// Archived legacy standalone mockup test.
import { expect, test } from '@playwright/test';
import path from 'node:path';
import { pathToFileURL } from 'node:url';

test('opens the standalone mockup directly from disk', async ({ page }) => {
  const errors: string[] = [];
  page.on('console', (message) => {
    if (message.type() === 'error') errors.push(message.text());
  });
  page.on('pageerror', (error) => errors.push(error.message));

  const mockupUrl = pathToFileURL(
    path.resolve('mockups/stringsight-marked-score.html'),
  ).toString();
  await page.goto(mockupUrl);

  await expect(page).toHaveTitle('StringSight — The Marked-Up Score');
  await page.waitForTimeout(100);
  expect(errors).toEqual([]);
  await expect(page.getByRole('heading', { name: '12—16' })).toBeVisible();
  await expect(page.getByRole('button', { name: 'Record a take' })).toBeVisible();
});
