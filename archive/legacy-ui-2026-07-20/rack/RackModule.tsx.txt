// Archived legacy rack module.
import type { ReactNode } from 'react';

import { RackStatus, type RackStatusTone } from './RackControls';

export type RackModuleSize = 'compact' | 'standard' | 'expanded';

export type RackModuleProps = {
  actions?: ReactNode;
  children: ReactNode;
  description?: string;
  moduleId: string;
  size?: RackModuleSize;
  status?: string;
  statusTone?: RackStatusTone;
  title: string;
  unit: string;
};

const screwPositions = ['top-left', 'top-right', 'bottom-left', 'bottom-right'] as const;

function RackModuleHardware() {
  return screwPositions.map((position) => (
    <span
      aria-hidden="true"
      className={`ss-rack-screw ss-rack-screw--${position}`}
      key={position}
    />
  ));
}

export function RackModule({
  actions,
  children,
  description,
  moduleId,
  size = 'standard',
  status,
  statusTone = 'idle',
  title,
  unit,
}: RackModuleProps) {
  const titleId = `${moduleId}-rack-title`;

  return (
    <section
      aria-labelledby={titleId}
      className={`ss-rack-module ss-rack-module--${size}`}
      data-rack-module={moduleId}
    >
      <RackModuleHardware />
      <header className="ss-rack-module-header">
        <div className="ss-rack-module-identity">
          <span className="ss-rack-module-index">{unit}</span>
          <h2 id={titleId}>{title}</h2>
          {description !== undefined && <p>{description}</p>}
        </div>
        {(status !== undefined || actions !== undefined) && (
          <div className="ss-rack-module-actions">
            {status !== undefined && <RackStatus tone={statusTone}>{status}</RackStatus>}
            {actions}
          </div>
        )}
      </header>
      <div className="ss-rack-module-content">{children}</div>
    </section>
  );
}
